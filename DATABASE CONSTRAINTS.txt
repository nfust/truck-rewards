CONSTRAINTS:
CREATE TABLE manager(email varchar(50) PRIMARY KEY, first varchar(50) NOT NULL, middle varchar(50) NOT NULL, last varchar(50) NOT NULL, phone varchar(50) NOT NULL, username varchar(50) UNIQUE NOT NULL, pass varchar(50) NOT NULL);


CREATE TABLE sponsor(email varchar(50) PRIMARY KEY, first varchar(50) NOT NULL, middle varchar(50) NOT NULL, last varchar(50) NOT NULL, phone varchar(50) NOT NULL, username varchar(50) UNIQUE NOT NULL, pass varchar(50) NOT NULL);




CREATE TABLE driver(email varchar(50) PRIMARY KEY, first varchar(50) NOT NULL, middle varchar(50) NOT NULL, last varchar(50) NOT NULL, phone varchar(50) NOT NULL, username varchar(50) UNIQUE NOT NULL, pass varchar(50) NOT NULL, points INTEGER NOT NULL);


CREATE TABLE user(email varchar(50) PRIMARY KEY, first varchar(50) , middle varchar(50), last varchar(50), company varchar(50) NOT NULL, type varchar(10) NOT NULL, phone varchar(50), username varchar(50) UNIQUE NOT NULL, pass varchar(50) NOT NULL, points INTEGER NOT NULL, overTime INTEGER NOT NULL);


create table payment(id integer PRIMARY KEY, user varchar(50) NOT NULL,productID INTEGER NOT NULL, status varchar(50) NOT NULL, time varchar(50) NOT NULL);






create table catalog(companyName varchar(50) PRIMARY KEY, price INTEGER NOT NULL, image varchar(150) NOT NULL, id INTEGER NOT NULL);
